<script>
    const API = '/api';
    function safeGetToken() {
        if (typeof getToken === 'function') return getToken();
        return localStorage.getItem('token') || '';
    }

    function safeGetUser() {
        if (typeof getUser === 'function') return getUser() || {};
        try {
            return JSON.parse(localStorage.getItem('user') || '{}');
        } catch {
            return {};
        }
    }

    const user = safeGetUser();
    const isAdmin = String(user.role || '').toLowerCase() === 'admin';

    function authHeaders() {
        const token = safeGetToken();
        return token ? { Authorization: `Bearer ${token}` } : {};
    }
    
    let allInvoices = [];

    async function load() {
        try {
            const res = await fetch(`${API}/sales`, { headers: authHeaders() });
            if (!res.ok) throw new Error('فشل تحميل البيانات');
            const data = await res.json();
            allInvoices = data;
            renderTable(data);
        } catch(e) { alert('خطأ: ' + e.message); }
    }

    function renderTable(data) {
        if (!data || data.length === 0) {
            document.getElementById('tbody').innerHTML = '<tr><td colspan="8" style="padding:20px;color:#666;">لا توجد فواتير مسجلة</td></tr>';
            return;
        }

        document.getElementById('tbody').innerHTML = data.map(i => {
            const isPaid = i.paymentStatus === 'Paid' || (i.paidAmount >= i.finalTotal);
            const carInfo = encodeURIComponent(i.carModel ? `${i.carModel.brand} ${i.carModel.model}` : '-');
            const custName = encodeURIComponent(i.customer?.name || '-');
            const custId = i.customer?.accountId || '';
            const payLink = `/financial/treasury.html?invoiceId=${i._id}&invNum=${i.invoiceNumber}&amount=${i.finalTotal}&type=Inbound&custId=${custId}&custName=${custName}&carInfo=${carInfo}`;
            const cutLink = `/inventory/cutting_planner.html?invoiceId=${i._id}`;
            // روابط الربط المباشر — رقم الفاتورة هو المفتاح المشترك
            const jobLink      = `/service/service_jobs.html`;
            const stockOutLink = `/inventory/stock_out.html?jobOrder=${encodeURIComponent(i.invoiceNumber)}`;

            return `
            <tr data-id="${i._id}">
                <td><input type="checkbox" class="row-select" value="${i._id}"></td>
                <td>${i.invoiceNumber}</td>
                <td>${new Date(i.date).toLocaleDateString('ar-EG')}</td>
                <td>${i.customer?.name || '-'}</td>
                <td>${i.carModel ? `${i.carModel.brand} ${i.carModel.model}` : '-'}</td>
                <td>${i.salesPersonName || '-'}</td>
                <td style="font-weight:bold; color:green">${parseFloat(i.finalTotal).toLocaleString()}</td>
                <td><span class="${isPaid?'paid-badge':'unpaid-badge'}">${isPaid?'مدفوع':'غير مدفوع'}</span></td>
                <td>
                    <button class="btn btn-cut" onclick="location.href='${cutLink}'">✂️ تقطيع</button>
                    ${!isPaid ? `<button class="btn btn-pay" onclick="location.href='${payLink}'">💰 سداد</button>` : ''}
                    ${isAdmin ? `<button class="btn btn-edit" onclick="location.href='/sales/sales_invoice.html?id=${i._id}'">✏️</button>` : ''}
                    <button class="btn" style="background:#8b5cf6; color:white;" onclick="location.href='/sales/sales_invoice_details.html?id=${i._id}'">👁️</button>
                    <button class="btn btn-print" onclick="printInvoice('${i._id}')">🖨️</button>
                    <button class="btn" style="background:#0f766e; color:white;" title="أمر التشغيل" onclick="location.href='${jobLink}'">📋</button>
                    <button class="btn" style="background:#b45309; color:white;" title="إذن الصرف - ${i.invoiceNumber}" onclick="location.href='${stockOutLink}'">📦</button>
                    ${isAdmin ? `<button class="btn btn-del" onclick="del('${i._id}')">🗑️</button>` : ''}
                </td>
            </tr>
            `;
        }).join('');
    }

    function filterTable() {
        const query = document.getElementById('searchInput').value.toLowerCase();
        const from = document.getElementById('dateFrom').value ? new Date(document.getElementById('dateFrom').value) : null;
        const to = document.getElementById('dateTo').value ? new Date(document.getElementById('dateTo').value) : null;
        if(to) to.setHours(23,59,59);

        const filtered = allInvoices.filter(i => {
            const invNum = String(i.invoiceNumber).toLowerCase();
            const custName = i.customer?.name ? i.customer.name.toLowerCase() : '';
            const invDate = new Date(i.date);

            // Text Match
            const matchText = invNum.includes(query) || custName.includes(query);
            
            // Date Match
            let matchDate = true;
            if(from && invDate < from) matchDate = false;
            if(to && invDate > to) matchDate = false;

            return matchText && matchDate;
        });
        renderTable(filtered);
    }

    function toggleSelectAll() {
        const checked = document.getElementById('selectAll').checked;
        document.querySelectorAll('.row-select').forEach(cb => cb.checked = checked);
    }

    async function deleteSelected() {
        if ( !isAdmin ) { alert('الحذف متاح للأدمن فقط'); return; }
        const selected = Array.from(document.querySelectorAll('.row-select:checked')).map(cb => cb.value);
        if (selected.length === 0) { alert('لم يتم تحديد أي فاتورة للحذف'); return; }
        
        if (!confirm(`تحذير خطير! ⚠️\n\nأنت على وشك حذف (${selected.length}) فاتورة نهائياً.\nسيتم عكس جميع تأثيرات المخزن والقيود المحاسبية.\n\nهل أنت متأكد تماماً؟`)) return;

        let successCount = 0;
        for (const id of selected) {
            try {
                const res = await fetch(`${API}/sales/${id}`, { method: 'DELETE', headers: authHeaders() });
                if(res.ok) successCount++;
            } catch(e) { console.error(e); }
        }
        
        alert(`تم حذف ${successCount} من أصل ${selected.length} فاتورة بنجاح.`);
        load();
    }

    async function del(id) {
        if ( !isAdmin ) { alert('الحذف متاح للأدمن فقط'); return; }
        if(!confirm('هل أنت متأكد من الحذف؟ سيتم إلغاء القيد وعكس المخزن.')) return;
        try {
            const res = await fetch(`${API}/sales/${id}`, { method: 'DELETE', headers: authHeaders() });
            if(res.ok) { alert('تم الحذف وتصحيح الأوضاع ✅'); load(); }
            else { 
                const err = await res.json();
                alert('فشل الحذف: ' + err.message); 
            }
        } catch(e) { alert('خطأ اتصال'); }
    }

    function printInvoice(id) {
        window.open('/sales/job_order_print.html?id=' + id, '_blank');
    }

    load();
</script>

<script>
    const API = '/api';
    
    window.onload = async () => {
        const id = new URLSearchParams(window.location.search).get('id');
        if(!id) return alert('لا يوجد رقم فاتورة');

        try {
            const token = localStorage.getItem('token'); 
            const res = await fetch(`${API}/sales/${id}`, {
                headers: { 'Authorization': `Bearer ${token}` }
            });
            if (!res.ok) throw new Error("فاتورة غير موجودة");
            const inv = await res.json();

            // Fill header information
            document.getElementById('invNum').innerText = inv.invoiceNumber;
            document.getElementById('date').innerText = new Date(inv.date).toLocaleDateString('ar-EG');
            document.getElementById('custName').innerText = inv.customer?.name || inv.customerName || '-';
            document.getElementById('phone').innerText = inv.customer?.phone || '-';
            document.getElementById('vehicleType').innerText = inv.vehicleType || '-';
            document.getElementById('carModel').innerText = inv.carModel || '-';
            document.getElementById('vehiclePlate').innerText = inv.vehiclePlate || '-';
            document.getElementById('vehicleColor').innerText = inv.vehicleColor || '-';
            document.getElementById('serviceType').innerText = inv.serviceType || '-';
            document.getElementById('salesPersonName').innerText = inv.salesPerson?.name || inv.salesPerson || '-';
            document.getElementById('notes').innerText = inv.notes || '-';
            
            // Fill items table
            const itemsBody = document.getElementById('itemsBody');
            let itemsHtml = '';
            
            (inv.items || []).forEach((item, index) => {
                const partName = item.partName || item.product?.name || 'خدمة';
                const productName = item.product?.name || item.productName || '-';
                const length = item.length || 0;
                const width = item.width || 0;
                const area = item.area || 0;
                const price = item.price || 0;
                const total = item.lineTotal || item.total || 0;
                
                itemsHtml += `
                    <tr>
                        <td>${index + 1}</td>
                        <td class="desc">${partName}</td>
                        <td>${productName}</td>
                        <td>${length}</td>
                        <td>${width}</td>
                        <td>${area.toFixed(4)}</td>
                        <td>${Number(price).toLocaleString('ar-EG', {minimumFractionDigits: 2})}</td>
                        <td class="total">${Number(total).toLocaleString('ar-EG', {minimumFractionDigits: 2})}</td>
                    </tr>
                `;
            });
            
            itemsBody.innerHTML = itemsHtml;
            
            // Fill financials
            const subtotal = parseFloat(inv.subtotal) || 0;
            const extraCost = parseFloat(inv.totalExtraCosts) || 0;
            const discount = parseFloat(inv.discount) || 0;
            const totalBeforeVat = subtotal + extraCost - discount;
            const vat = totalBeforeVat * 0.14;
            const finalTotal = totalBeforeVat + vat;
            
            document.getElementById('subtotal').innerText = subtotal.toLocaleString('ar-EG', {minimumFractionDigits: 2});
            document.getElementById('extraCost').innerText = extraCost.toLocaleString('ar-EG', {minimumFractionDigits: 2});
            document.getElementById('discount').innerText = discount.toLocaleString('ar-EG', {minimumFractionDigits: 2});
            document.getElementById('totalBeforeVat').innerText = totalBeforeVat.toLocaleString('ar-EG', {minimumFractionDigits: 2});
            document.getElementById('vatVal').innerText = vat.toLocaleString('ar-EG', {minimumFractionDigits: 2});
            document.getElementById('finalTotal').innerText = finalTotal.toLocaleString('ar-EG', {minimumFractionDigits: 2});
            
        } catch(e) {
            alert('خطأ في تحميل الفاتورة: ' + e.message);
            console.error('Invoice Load Error:', e);
        }
    };
</script>
