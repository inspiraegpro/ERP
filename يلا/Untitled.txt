def get_data(query, params=()):
    try:
        conn = get_connection(); c = conn.cursor()

        conn.close(); return data

def get_last_serial(prefix):
    try:
        conn = get_connection(); c = conn.cursor()
        c.execute(f"SELECT id FROM inventory ORDER BY id DESC LIMIT 1")
        last = c.fetchone()
        conn.close()
        num = (last[0] + 1) if last else 1
        return f"{prefix}-{datetime.now().strftime('%y%m')}{num:04d}"
    except: return f"{prefix}-0001"
